typescriptconst APIFY_TOKEN = process.env.APIFY_API_TOKEN;

// Transcribe a single Instagram reel by URL
export async function transcribeReel(reelUrl: string): Promise<{ transcript: string; language: string } | null> {
  if (!APIFY_TOKEN) return null;
  
  try {
    const response = await fetch(
      'https://api.apify.com/v2/acts/bulletproof~instagram-transcript-extractor/run-sync-get-dataset-items?token=' + APIFY_TOKEN,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          urls: [reelUrl],
          format: 'json',
          includeMetadata: false
        })
      }
    );
    
    if (!response.ok) {
      console.error('Apify transcription failed:', response.status);
      return null;
    }
    
    const results = await response.json();
    if (results && results.length > 0) {
      const item = results[0];
      // The actor returns fullText or segments depending on format
      const transcript = item.fullText || item.text || 
        (item.segments ? item.segments.map((s: any) => s.text).join(' ') : '');
      return {
        transcript: transcript.trim(),
        language: item.language || 'es'
      };
    }
    return null;
  } catch (error) {
    console.error('Transcription error:', error);
    return null;
  }
}

// Batch transcribe multiple reels
export async function transcribeReels(reelUrls: string[]): Promise<Map<string, string>> {
  const results = new Map<string, string>();
  
  for (const url of reelUrls) {
    const result = await transcribeReel(url);
    if (result && result.transcript) {
      results.set(url, result.transcript);
    }
    // Small delay between calls to avoid rate limiting
    await new Promise(resolve => setTimeout(resolve, 500));
  }
  
  return results;
}
New API route: POST /api/transcribe
Body: { reelId?: number, url?: string, batch?: boolean }
Single reel mode (reelId or url provided):

Get the reel from DB (or use provided URL)
Call transcribeReel(permalink)
Store transcript in the reels table transcripcion field
Return: { success: true, transcript: string }

Batch mode (batch: true):

Get all reels from DB where transcripcion IS NULL AND permalink IS NOT NULL
For each, call transcribeReel(permalink)
Store transcripts in DB
Return: { transcribed: number, failed: number, errors: string[] }

New API route: POST /api/transcribe/competitor
Body: { competitorReelId: number }

Get the competitor reel from competitor_reels table
Call transcribeReel(url)
Store in competitor_reels.transcripcion field
Return: { success: true, transcript: string }

New API route: POST /api/transcribe/competitor-batch
Body: { competitorId: number } (transcribe all reels from one competitor)

Get all competitor_reels for this competitor where transcripcion IS NULL AND url IS NOT NULL
Transcribe each
Store results
Return: { transcribed: number, failed: number }
